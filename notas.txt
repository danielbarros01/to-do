1.npx sequelize-cli init
2.npx sequelize-cli db:migrate

//Crear modelos
npx sequelize-cli model:generate --name Task --attributes title:string,description:string,priority:string,status:string,expiration_date:date,creation_date:date,date_of_resolution:date

//Crear Seeds
npx sequelize-cli seed:generate --name create-lists
npx sequelize-cli seed:generate --name more-users
npx sequelize-cli seed:generate --name create-some-addresses
npx sequelize-cli seed:generate --name mucho-more-users

//Seed(Enviar datos a la BD)
npx sequelize-cli db:seed:all

//Deshacer Seeds
npx sequelize-cli db:seed:undo:all
//Deshacer especifico
npx sequelize-cli db:seed:undo --seed 20221017221614-more-users.js

//Migracion
npx sequelize-cli db:migrate
//Crear Migracion
npx sequelize-cli migration:generate --name create-users-table
//Deshacer ultima Migracion
npx sequelize-cli db:migrate:undo
//Deshacer todas las migraciones
npx sequelize-cli db:migrate:undo:all

//Agregar campos con migracion
npx sequelize-cli migration:generate --name add-city-and-country-to-addresses-table
